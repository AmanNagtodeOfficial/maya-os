# Maya OS - Ubuntu-like Desktop Interface

This document outlines the implementation of an Ubuntu-like user interface for Maya OS, featuring components similar to Unity and GNOME desktop environments.

## Desktop Environment Components

### 1. Desktop Manager (include/gui/desktop.h)

```c
#ifndef DESKTOP_H
#define DESKTOP_H

#include <stdint.h>
#include "gui/window.h"
#include "gui/graphics.h"

    int x, y;
        int width, height;
            char *name;
                char *path;
                    uint8_t *icon_data;
                        uint8_t selected;
                        } desktop_icon_t;

                        typedef struct {
                            desktop_icon_t icons[MAX_DESKTOP_ICONS];
                                            ```

                                                                                                                                                                                                                                                                                                                                                                                                                    static void desktop_create_gradient_wallpaper(void) {
                                                                                                                                                                                                                                                                                                                                                                                                                        // Create Ubuntu-style purple to orange gradient
                                                                                                                                                                                                                                                                                                                                                                                                                            for (int y = 0; y < SCREEN_HEIGHT; y++) {
                                                                                                                                                                                                                                                                                                                                                                                                                                    for (int x = 0; x < SCREEN_WIDTH; x++) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                // Calculate gradient color (purple to orange)
                                                                                                                                                                                                                                                                                                                                                                                                                                                            float ratio = (float)y / SCREEN_HEIGHT;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                        uint8_t r = (uint8_t)(0x4C * (1 - ratio) + 0xFF * ratio);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    uint8_t g = (uint8_t)(0x1D * (1 - ratio) + 0x8C * ratio);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                uint8_t b = (uint8_t)(0x4F * (1 - ratio) + 0x00 * ratio);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        // Convert to 8-bit color
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    uint8_t color = graphics_rgb_to_color(r, g, b);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                graphics_set_pixel(x, y, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            }
