                                                                                                                                                                                                                                                                                                                                    void ubuntu_graphics_init(void) {
                                                                                                                                                                                                                                                                                                                                        // Initialize VGA graphics mode
                                                                                                                                                                                                                                                                                                                                            vga_set_mode_13h(); // 320x200x256 colors
                                                                                                                                                                                                                                                                                                                                            }

                                                                                                                                                                                                                                                                                                                                            void ubuntu_set_pixel(int x, int y, uint8_t color) {
                                                                                                                                                                                                                                                                                                                                                if (x >= 0 && x < SCREEN_WIDTH && y >= 0 && y < SCREEN_HEIGHT) {
                                                                                                                                                                                                                                                                                                                                                        vga_put_pixel(x, y, color);
                                                                                                                                                                                                                                                                                                                                                            }
                                                                                                                                                                                                                                                                                                                                                            }

                                                                                                                                                                                                                                                                                                                                                            uint8_t ubuntu_get_pixel(int x, int y) {
                                                                                                                                                                                                                                                                                                                                                                if (x >= 0 && x < SCREEN_WIDTH && y >= 0 && y < SCREEN_HEIGHT) {
                                                                                                                                                                                                                                                                                                                                                                        uint8_t *vga_memory = (uint8_t *)0xA0000;
                                                                                                                                                                                                                                                                                                                                                                                return vga_memory[y * SCREEN_WIDTH + x];
                                                                                                                                                                                                                                                                                                                                                                                    }
                                                                                                                                                                                                                                                                                                                                                                                        return 0;
                                                                                                                                                                                                                                                                                                                                                                                        }

                                                                                                                                                                                                                                                                                                                                                                                        void ubuntu_fill_rect(int x, int y, int width, int height, uint8_t color) {
                                                                                                                                                                                                                                                                                                                                                                                            for (int j = 0; j < height; j++) {
                                                                                                                                                                                                                                                                                                                                                                                                    for (int i = 0; i < width; i++) {
                                                                                                                                                                                                                                                                                                                                                                                                                ubuntu_set_pixel(x + i, y + j, color);
                                                                                                                                                                                                                                                                                                                                                                                                                        }
                                                                                                                                                                                                                                                                                                                                                                                                                            }
                                                                                                                                                                                                                                                                                                                                                                                                                            }

                                                                                                                                                                                                                                                                                                                                                                                                                            void ubuntu_draw_rect(int x, int y, int width, int height, uint8_t color) {
                                                                                                                                                                                                                                                                                                                                                                                                                                // Top and bottom lines
                                                                                                                                                                                                                                                                                                                                                                                                                                    for (int i = 0; i < width; i++) {
                                                                                                                                                                                                                                                                                                                                                                                                                                            ubuntu_set_pixel(x + i, y, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                    ubuntu_set_pixel(x + i, y + height - 1, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                        }
                                                                                                                                                                                                                                                                                                                                                                                                                                                            // Left and right lines
                                                                                                                                                                                                                                                                                                                                                                                                                                                                for (int j = 0; j < height; j++) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                        ubuntu_set_pixel(x, y + j, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                ubuntu_set_pixel(x + width - 1, y + j, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    }

                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    void ubuntu_draw_line(int x1, int y1, int x2, int y2, uint8_t color) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        int dx = x2 - x1;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            int dy = y2 - y1;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                int steps = (dx < 0 ? -dx : dx) + (dy < 0 ? -dy : dy);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        if (steps == 0) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                ubuntu_set_pixel(x1, y1, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        return;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    float x_inc = (float)dx / steps;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        float y_inc = (float)dy / steps;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                float x = x1;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    float y = y1;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            for (int i = 0; i <= steps; i++) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    ubuntu_set_pixel((int)x, (int)y, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            x += x_inc;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    y += y_inc;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        }

                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        void ubuntu_fill_circle(int cx, int cy, int radius, uint8_t color) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            for (int y = -radius; y <= radius; y++) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    for (int x = -radius; x <= radius; x++) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                if (x * x + y * y <= radius * radius) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                ubuntu_set_pixel(cx + x, cy + y, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        }

                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        void ubuntu_draw_circle(int cx, int cy, int radius, uint8_t color) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            int x = 0;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                int y = radius;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    int d = 3 - 2 * radius;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            while (y >= x) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    ubuntu_set_pixel(cx + x, cy + y, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            ubuntu_set_pixel(cx - x, cy + y, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    ubuntu_set_pixel(cx + x, cy - y, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            ubuntu_set_pixel(cx - x, cy - y, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    ubuntu_set_pixel(cx + y, cy + x, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            ubuntu_set_pixel(cx - y, cy + x, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    ubuntu_set_pixel(cx + y, cy - x, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            ubuntu_set_pixel(cx - y, cy - x, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            x++;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    if (d > 0) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                y--;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            d = d + 4 * (x - y) + 10;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    } else {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                d = d + 4 * x + 6;
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            }

                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    ubuntu_set_pixel(x + col, y + row, color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                } else if (bg_color != 0) {
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                ubuntu_set_pixel(x + col, y + row, bg_color);
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                            }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                    }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        }
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        }

                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                        void ubuntu_draw_string(int x, int y, const char* str, uint8_t color, uint8_t bg_color) {
